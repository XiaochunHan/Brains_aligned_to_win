setwd('../../Data/Develop')

dataset_BOLD <- read.csv('BOLD_single_trial_outcome.csv',header=TRUE,sep=",")
dataset_FC <- read.csv('FC_single_trial_outcome.csv',header=TRUE,sep=",")
dataset_BINS <- read.csv('BINS_single_trial_outcome.csv',header=TRUE,sep=",")
dataset_INS <- read.csv('INS_single_trial_outcome.csv',header=TRUE,sep=",")
dataset_y <- read.csv('win_lose_and_pay_single_trial.csv',header=TRUE,sep=",")
dataset_y_attack <- dataset_y[dataset_y$role.A.1. == 1,]
dataset_y_attack <- dataset_y_attack[-c(4,14,22),] # Three sessions are deleted due to the absent of winning trial(4,22) and brain signal(14). 

BOLD_mean = extract_win_or_pay_mean(dataset_BOLD,dataset_y_attack,"BOLD_","win",FALSE)
FC_mean = extract_win_or_pay_mean(dataset_FC,dataset_y_attack,"FC_","win",FALSE)
INS_mean = extract_win_or_pay_mean(dataset_INS,dataset_y_attack,"WNS_","win",FALSE)
BINS_mean = extract_win_or_pay_mean(dataset_BINS,dataset_y_attack,"BNS_","win",FALSE)
data_all_outcome = cbind(INS_mean[,c(1,3:ncol(INS_mean))],BINS_mean[,3:ncol(BINS_mean)],BOLD_mean[,3:ncol(BOLD_mean)],FC_mean[,3:ncol(FC_mean)])
data_INS_BINS_outcome = cbind(INS_mean[,c(1,3:ncol(INS_mean))],BINS_mean[,3:ncol(BINS_mean)])
data_BOLD_FC_outcome = cbind(BOLD_mean[,c(1,3:ncol(BOLD_mean))],FC_mean[,3:ncol(FC_mean)])

data_WNS_outcome = INS_mean[,c(1,3:ncol(INS_mean))]
data_BNS_outcome = BINS_mean[,c(1,3:ncol(BINS_mean))]
data_BOLD_outcome = BOLD_mean[,c(1,3:ncol(BOLD_mean))]
data_FC_outcome = FC_mean[,c(1,3:ncol(FC_mean))]

data_minus_WNS_outcome = cbind(BINS_mean[,c(1,3:ncol(BINS_mean))],BOLD_mean[,3:ncol(BOLD_mean)],FC_mean[,3:ncol(FC_mean)])
data_minus_BNS_outcome = cbind(INS_mean[,c(1,3:ncol(INS_mean))],BOLD_mean[,3:ncol(BOLD_mean)],FC_mean[,3:ncol(FC_mean)])
data_minus_BOLD_outcome = cbind(INS_mean[,c(1,3:ncol(INS_mean))],BINS_mean[,3:ncol(BINS_mean)],FC_mean[,3:ncol(FC_mean)])
data_minus_FC_outcome = cbind(INS_mean[,c(1,3:ncol(INS_mean))],BINS_mean[,3:ncol(BINS_mean)],BOLD_mean[,3:ncol(BOLD_mean)])

acc_all_perm = readRDS("SVM_win_data_all_outcome_phase_perm_5000.RData")
acc_all_low_25 = sort(acc_all_perm$acc_perm)[125]
acc_all_top_25 = sort(acc_all_perm$acc_perm)[4875]
acc_all_perm$ci95 = rep(0,5000)
acc_all_perm$ci95[(acc_all_perm$acc_perm<acc_all_low_25|acc_all_perm$acc_perm>acc_all_top_25)] = 1
p_all_perm = mean(acc_all<acc_all_perm$acc_perm)
print(p_all_perm)
## [1] 0
p<-ggplot(acc_all_perm, aes(x = acc_perm, fill = as.factor(ci95))) +
